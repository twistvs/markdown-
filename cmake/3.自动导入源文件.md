**1.使用`file`命令**

```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(CMAKE_CXX_STANDARD 17)

file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)

#add_executable(可执行文件名称 源文件1 源文件2 ...)
add_executable(Cmake_Learning ${SRC})
```

`CMAKE_CURRENT_SOURCE_DIR`是CMakeLists.txt所在的目录，通过*/.cpp可以查找目录中所有.cpp文件并存到SRC中
