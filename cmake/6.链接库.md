```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(CMAKE_CXX_STANDARD 17)

file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)
# set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)

#add_executable(可执行文件名称 源文件1 源文件2 ...)
add_executable(Cmake_Learning ${SRC})
# add_library(Cmake_Learning SHARED ${SRC})    #STATIC静态库，SHARED动态库

target_include_directories(Cmake_Learning PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)

target_link_libraries(Cmake_Learning PUBLIC calculate)
target_link_directories(Cmake_Learning PUBLIC ${PROJECT_SOURCE_DIR}/lib)
```

由于已经把函数做成了库，所以只需要main.cpp文件

<u>链接静态库需要</u>

1. 库的头文件，用`target_include_directories`指定头文件的位置
2. 库的名称，在制作库时，`add_library`中的名称就是库名称，也是库文件Lib?.a中的?。用`target_link_libraries`指定链接的库的名称
3. 库的位置，用`target_link_directories`指定连接的库的位置
