**1.用`set`打包源文件的路径**

```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(SRC main.cpp add.cpp sub.cpp)
#add_executable(可执行文件名称 源文件1 源文件2 ...)
add_executable(Cmake_Learning ${SRC})
```

- cmake命令==无法==在已经拥有cmake相关文件的目录运行
- 如果修改了源文件的内容，而没有修改CMakeLists.txt，则不需要再次运行cmake，只需要重新运行make

**2.用`set`设置c++的版本**

```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(CMAKE_CXX_STANDARD 17)

set(SRC main.cpp add.cpp sub.cpp)

set()
#add_executable(可执行文件名称 源文件1 源文件2 ...)
```

**3.用`set`设置可执行文件的输出路径**

```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(CMAKE_CXX_STANDARD 17)

set(SRC main.cpp add.cpp sub.cpp)

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)

#add_executable(可执行文件名称 源文件1 源文件2 ...)
add_executable(Cmake_Learning ${SRC})
```

PROJECT_SOURCE_DIR是项目的根目录
