```cmake
#这是一个cmake项目的CMakeLists.txt文件

cmake_minimum_required(VERSION 3.12)

project(Cmake_Learning)

set(CMAKE_CXX_STANDARD 17)

file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)

# set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)

#add_executable(可执行文件名称 源文件1 源文件2 ...)
# add_executable(Cmake_Learning ${SRC})
add_library(calculate SHARED ${SRC})    #STATIC静态库，SHARED动态库

target_include_directories(Cmake_Learning PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
```

制作==库文件==与制作==可执行文件==是类似的

|          |                      制作库文件                      |                     制作可执行文件                      |
| :------- | :--------------------------------------------------: | :-----------------------------------------------------: |
| 输出目录 | `set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)` | `set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)` |
| 执行项目 |        `add_library(calculate SHARED ${SRC})`        |         `add_executable(Cmake_Learning ${SRC})`         |
