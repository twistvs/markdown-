二叉树的数据结构

```c++
struct treeNode
{
	int val;
	treeNode* lchild;
	treeNode* rchild;
	treeNode(int val)
	{
		this->val = val;
		lchild = nullptr;
		rchild = nullptr;
	}
};
```

建立二叉搜索树

```c++
//建立二叉搜索树
treeNode* build_binerySearchTree(std::vector<int>& vec)
{
	auto root = new treeNode(vec[0]);
	//按照vector的顺序循环，把所有元素一个个插到树里面
	for (int i = 1; i < vec.size(); i++)
	{
		auto troot = root;
		//对于每一个元素，都需要循环与各个节点比较大小，判断自己的插入位置，用troot来记录元素现在比较到了哪个节点
		while (troot != nullptr)
		{
			if (vec[i] < troot->val)
			{
				//如果元素小于troot，则元素在troot左边，如果troot没有左儿子，那就插到这里；如果有左儿子，则继续和左儿子比，troot转移到左儿子
				if (troot->lchild == nullptr)
				{
					//当前元素已经插入完毕，直接跳到下一个元素
					troot->lchild = new treeNode(vec[i]);
					break;
				}
				else
				{
					troot = troot->lchild;
				}
			}
			if (vec[i] > troot->val)
			{
				if (troot->rchild == nullptr)
				{
					troot->rchild = new treeNode(vec[i]);
					break;
				}
				else
				{
					troot = troot->rchild;
				}
			}
		}
	}
	return root;
}
```

通过递归实现前中后序遍历

```c++
//中序遍历，左中右
void InOrder(treeNode* root)
{
	if (root == nullptr)
	{
		return;
	}
	InOrder(root->lchild);
	std::cout << root->val << ",";
	InOrder(root->rchild);
}

//前序遍历，中左右
void PreOrder(treeNode* root)
{
	if (root == nullptr)
	{
		return;
	}
	std::cout << root->val << ",";
	InOrder(root->lchild);
	InOrder(root->rchild);
}

//后序遍历，左右中
void PostOrder(treeNode* root)
{
	if (root == nullptr)
	{
		return;
	}
	InOrder(root->lchild);
	InOrder(root->rchild);
	std::cout << root->val << ",";
}
```

基于广度优先搜索实现的层次遍历

```c++
void LayerOrder(treeNode* root)
{
	//利用队列先进先出的特性进行层次遍历
	std::queue<treeNode*> q;
	q.push(root);
	//last是当前行的最后一个元素，第一行的最后一个元素是root;nextLast是下一行的最后一个元素，经过每次循环来计算，并动态更新last为nextLast，就可以把每一行的最后一个元素记录下来
	treeNode* last = root;
	treeNode* nextLast = NULL;
	//如果为空会返回0
	while (!q.empty())
	{
		treeNode* tmp = q.front();
		std::cout << tmp->val;
		q.pop();
		if (tmp->lchild)
		{
			//nextLast随着下一行的元素动态更新，当tmp走到最后一个元素时，nextLast正好走到下一行的最后一个元素
			q.push(tmp->lchild);
			nextLast = tmp->lchild;
		}
		if (tmp->rchild)
		{
			q.push(tmp->rchild);
			nextLast = tmp->rchild;
		}
		if (tmp == last)
		{
			std::cout << std::endl;
			last = nextLast;
		}
	}
}
```

查找算法

时间复杂度O(logn)

```c++
bool find(treeNode* root, int target)
{
	while (root)
	{
        //与root比较，如果与root相等，则查找成功，如果比root小，就与root的左儿子比，依次类推
		if (target == root->val)
		{
			return true;
		}
		if (target < root->val)
		{
			root = root->lchild;
		}
		if (target > root->val)
		{
			root = root->rchild;
		}
	}
}
```
