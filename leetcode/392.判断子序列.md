#### 双指针

```c++
class Solution {
public:
    bool isSubsequence(string s, string t) {
        int sSize = s.size();
        int tSize = t.size();

        int sLeft = 0;
        int tLeft = 0;

        //如果相等，则sLeft和tLeft都走一步；如果不相等则只有t走一步
        while(sLeft < sSize && tLeft < tSize){
            if(s[sLeft] == t[tLeft]){
                sLeft++;
            }
            tLeft++;
        }

        //如果循环结束后，sLeft已经走到了sSize，则说明s中所有元素都有相等的
        if(sLeft == sSize){
            return true;
        }
        else{
            return false;
        }
    }
};
```
