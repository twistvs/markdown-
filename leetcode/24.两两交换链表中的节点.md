- 链表操作需要注意两个问题：

1. 一定要在前面添加dummy节点
2. nextNode一定要在判断curr不是null之后才能赋值，否则curr->next会报错

```c++
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummy = new ListNode(0);
        ListNode* pre = dummy;
        ListNode* curr = head;
        pre->next = head;

        while(pre->next && pre->next->next){
            //nextNode一定要在判断pre->next && pre->next->next都存在的情况下才能赋值，否则如果curr是null，那么curr->next就会报错
            ListNode* nextNode = curr->next;
            pre->next = nextNode;
            curr->next = nextNode->next;
            nextNode->next = curr;
            //pre和curr可以在换完之后就赋值，但是nextNode必须在判断完之后才能赋值
            pre = curr;
            curr = curr->next;
        }

        //注意不能返回head，因为第一个和第二个节点交换后，head变成第二个节点而不是第一个了
        return dummy->next;
    }
};
```
