#### 动态规划

`dp[i][j]`表示以`matrix[i][j]`为最右下角元素正方形的最大边长

**边界条件**：当i或者j为0时，如果此处值为1，则`dp[i][j] = 1`

**公式**:当此处值为1，则dp等于其左，上，左上三个dp的最小值+1

```c++
class Solution {
public:
    int maximalSquare(vector<vector<char>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();

        int ans = 0;

        vector<vector<int>> dp(m, vector<int>(n, 0));

        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(i == 0 || j == 0){
                    if(matrix[i][j] == '1'){
                        dp[i][j] = 1;
                    }
                }
                else{
                    if(matrix[i][j] == '1'){
                        dp[i][j] = min({dp[i - 1][j - 1], dp[i - 1][j], dp[i][j - 1]}) + 1;
                    }
                }
                ans = max(ans, dp[i][j]);
            }
        }

        return ans * ans;
    }
};
```
