#### 先改再验证（效率很低）

```c++
class Solution {
public:
    bool isPalindrome(string s) {
        int n = s.size();

        string s2;

        //把非字母数字字符的部分去掉，再把大写字母转换为小写字母
        for(char c : s){
            if(isdigit(c)){
                s2 += c;
            }
            if(isalpha(c)){
                if(c >= 65 && c <= 90){
                    s2 += (c + 32);
                }
                else{
                    s2 += c;
                }
            }
        }

        string s3 = s2;
        reverse(s3.begin(), s3.end());

        if(s2 == s3){
            return true;
        }
        else{
            return false;
        }
    }
};
```

#### 双指针

- 双指针只比较数字字母字符，非数字字母字符直接跳过
- 如果两者不同且都是字母，还需要判断他们是不是互为大小写关系

```c++
class Solution {
public:
    bool isPalindrome(string s) {
        int n = s.size();

        int left = 0;
        int right = n - 1;

        //在left和right相遇之前，只要不return false，那就说明是回文串
        while(left < right){
            //跳过非字母数字字符的部分
            while(!isalnum(s[left]) && left < right){
                left++;
            }
            while(!isalnum(s[right]) && left < right){
                right--;
            }
            
            if(s[left] != s[right]){
                //如果两者不同且都是字母，则需要判断他们是否是大小写的关系
                if(isalpha(s[left]) && isalpha(s[right])){
                    if(abs(s[left] - s[right]) != 32){
                        return false;
                    }
                }
                //如果两者不同且不是都为字母，则return false
                else{
                    return false;
                }
            }
            left++;
            right--;
        }

        return true;
    }
};
```
