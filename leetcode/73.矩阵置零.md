#### 用矩阵本身来记录标记

- 如果单独开一个空间用来记录哪一行哪一列需要置零，逻辑会非常简单
- 为了不额外开空间，我们要用矩阵本身的空间来记录哪一行哪一列需要置零

1. 使用第一行和第一列的元素来记录第i行/第j列是否需要置零
2. 由于第一行和第一列做了标志位，所以需要提前记录他们俩是否置零，但是这样只需要两个int，也就是O(1)的额外空间

```c++
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();

        //记录第一行和第一列里面有没有0，最后要用这个flag来判断第一行和第一列是否应该置零
        int flag_row0 = 0;
        int flag_col0 = 0;

        //判断第一行有没有0
        for(int j = 0; j < n; j++){
            if(matrix[0][j] == 0){
                flag_row0 = 1;
                break;
            }
        }

        //判断第一列有没有0
        for(int i = 0; i < m; i++){
            if(matrix[i][0] == 0){
                flag_col0 = 1;
                break;
            }
        }

        //如果第i行有0，那么让[i][0]置零，代表第i行有0；列同理
        //由于第一列和第一行是否置零由flag判断，所以这里的标志位不需要包含第一行和第一列
        for(int i = 1; i < m; i++){
            for(int j = 1; j < n; j++){
                if(matrix[i][j] == 0){
                    matrix[0][j] = 0;
                    matrix[i][0] = 0;
                }
            }
        }

        //如果[i][0]为0，说明第i行需要置零
        for(int i = 1; i < m; i++){
            if(matrix[i][0] == 0){
                for(int j = 0; j < n; j++){
                    matrix[i][j] = 0;
                }
            }
        }

        //如果[0][j]为0，说明第j列需要置零
        for(int j = 1; j < n; j++){
            if(matrix[0][j] == 0){
                for(int i = 0; i < m; i++){
                    matrix[i][j] = 0;
                }
            }
        }

        //最后通过flag来判断第一行和第一列是否置零
        if(flag_row0 == 1){
            for(int j = 0; j < n; j++){
                matrix[0][j] = 0;
            }
        }
        if(flag_col0 == 1){
            for(int i = 0; i < m; i++){
                matrix[i][0] = 0;
            }
        }
    }
};
```
