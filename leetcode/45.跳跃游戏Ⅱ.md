#### 贪心（难）

```c++
class Solution {
public:
    int jump(vector<int>& nums) {
        if (nums.size() <= 1) return 0;

        int steps = 0;          // 跳跃次数
        int curDistance = 0;    // 当前跳跃覆盖的最远距离下标
        int nextDistance = 0;   // 下一步跳跃能覆盖的最远距离下标

        // 注意：遍历范围是 0 到 n-2
        // 因为如果当前已经覆盖了最后一个元素，就不需要再多跳一次了
        for (int i = 0; i < nums.size() - 1; i++) {
            // 更新下一步能跳的最远位置
            nextDistance = max(nextDistance, i + nums[i]);

            // 当走到当前这一跳的“极限”时
            if (i == curDistance) {
                steps++;                // 必须跳一次
                curDistance = nextDistance; // 更新当前边界
                
                // 如果当前边界已经覆盖终点，可以提前跳出
                if (curDistance >= nums.size() - 1) break;
            }
        }
        return steps;
    }
};
```
