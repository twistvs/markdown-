- 对于每个点来说，他能接的雨水值等于`min(他左边最高的柱子，他右边最高的柱子)-他自己`，所以需要维护左边最高的柱子和右边最高的柱子
- 由于能接的雨水值只取决于两个柱子里更矮的那一个，所以总是移动更矮的那一边
- 在边界移动之前，我们需要确定这个边界是否能够接水和是否是新的最高边界

```c++
class Solution {
public:
    int trap(vector<int>& height) {
        int n = height.size();
        if (n <= 2) return 0;
        
        int left = 0, right = n - 1;
        int left_max = 0, right_max = 0;
        int ans = 0;
        
        while (left < right) {
            // 谁小移动谁，因为水位是由短板决定的
            if (height[left] < height[right]) {
                //由于left比左最高更高，所以很明显他是接不了水的
                if (height[left] >= left_max) {
                    left_max = height[left]; // 更新左侧最高墙
                } 
                //只有比左边界更矮才能接水
                else {
                    ans += (left_max - height[left]); // 计算当前柱子能接的水
                }
                left++;
            } else {
                if (height[right] >= right_max) {
                    right_max = height[right]; // 更新右侧最高墙
                } else {
                    ans += (right_max - height[right]); // 计算当前柱子能接的水
                }
                right--;
            }
        }
        return ans;
    }
};
```
