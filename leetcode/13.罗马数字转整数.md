```c++
class Solution {
public:
    int romanToInt(string s) {
        int n = s.size();

        //记录罗马数字与数字的映射
        unordered_map<char, int> m = {{'I', 1}, {'V', 5}, {'X', 10}, {'L', 50}, {'C', 100}, {'D', 500}, {'M', 1000}};

        int ans = 0;

        //由于罗马数字的正负由后面的罗马数字决定，所以反着遍历比较好
        //创建一个类似于链表哑节点的post，他的位置在s的后面，他的值需要小于所有罗马数字
        int post = 0;
        for(int i = n - 1; i >= 0; i--){
            int pre = m[s[i]];
            if(pre >= post){
                ans += pre;
            }
            else{
                ans -= pre;
            }
            post = pre;
        }

        return ans;
    }
};
```
