```c++
class Solution {
public:
    int lengthOfLastWord(string s) {
        int length = 0;

        int n = s.size();

        //从右边遍历，寻找最后一个单词的结尾；注意s可能全是空格，这样就会一路遍历到0
        int start = n - 1;
        while(s[start] == ' ' && start >= 0){
            start--;
        }

        //如果s不是全是空格，start最终会停在>=0的位置，从这里为起点反向遍历，获得最后一个单词的长度
        if(start >= 0){
            for(int i = start; i >= 0; i--){
                if(s[i] != ' '){
                    length++;
                }
                else{
                    break;
                }
            }
        }

        return length;
    }
};
```
