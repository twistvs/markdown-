#### BFS层次遍历

- 依次输出每一行的最右元素即可

```c++
class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        vector<int> ans;

        if(root == NULL){
            return ans;
        }
        queue<TreeNode*> q;

        q.push(root);

        while(!q.empty()){
            int size = q.size();
            for(int i = 0; i < size; i++){
                TreeNode* out = q.front();
                q.pop();
                if(i == size - 1){
                    ans.push_back(out->val);
                }
                if(out->left){
                    q.push(out->left);
                }
                if(out->right){
                    q.push(out->right);
                }
            }
        }

        return ans;
    }
};
```
