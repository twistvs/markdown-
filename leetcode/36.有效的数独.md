关键在于找到一个高效的存储数字num是否在某一行/列/块中出现过的方法

#### 哈希表

- 由于这道题的矩阵是定长的，所以用array稍微比vector高效一些
- 用`row[i][j]`来代表第i行是否出现过数字j，另外两个同理
- 元素`[i][j]`的块索引是`[i/3][j/3]`,转换为一维索引就是`(i / 3) * 3 + (j / 3)`

```c++
class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        //第i行是否出现过j
        array<array<bool, 9>, 9> row = {};
        //第i列是否出现过j
        array<array<bool, 9>, 9> col = {};
        //第i个九宫格是否出现过j
        array<array<bool, 9>, 9> box = {};

        for(int i = 0; i < 9; i++){
            for(int j = 0; j < 9; j++){
                char val = board[i][j];
        
                if(val == '.'){
                    continue;
                }
        
                int num = val - '1';
                int boxIndex = (i / 3) * 3 + (j / 3);
				
                //只要曾经出现过，又出现第二次，那就直接返回false
                if(row[i][num] || col[j][num] || box[boxIndex][num]){
                    return false;
                }

                row[i][num] = true;
                col[j][num] = true;
                box[boxIndex][num] = true;
            }
        }

        return true;
    }
};
```
