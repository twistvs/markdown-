#### 动态规划

- 如果最后一个元素相等，则`dp[i][j] = dp[i - 1][j - 1]`；如果不相等，则`dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])`

```c++
class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        int n1 = text1.size();
        int n2 = text2.size();

        //dp[i][j]表示text1的前i个元素和text2的前j个元素的最长公共子序列的长度
        vector<vector<int>> dp(n1 + 1, vector<int>(n2 + 1, 0));

        for(int i = 1; i <= n1; i++){
            char c1 = text1[i - 1];
            for(int j = 1; j <= n2; j++){
                char c2 = text2[j - 1];
                if(c1 == c2) dp[i][j] = dp[i - 1][j - 1] + 1;
                else  dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
            }
        }

        return dp[n1][n2];
    }
};
```
