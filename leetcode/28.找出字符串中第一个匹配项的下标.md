#### 暴力检查

```c++
class Solution {
public:
    int strStr(string haystack, string needle) {
        int needleSize = needle.size();
        int haystackSize = haystack.size();

        //依次检查每个子串就可以
        for(int i = 0; i <= haystackSize - needleSize; i++){
            if(haystack.substr(i, needleSize) == needle){
                return i;
            }
        }

        return -1;
    }
};
```
