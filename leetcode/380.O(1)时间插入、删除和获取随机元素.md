#### vector实现O(1)随机访问，map实现O(1)插入删除

```c++
class RandomizedSet {
public:
    RandomizedSet() {
    }
    
    //插入很简单
    bool insert(int val) {
        if(index.count(val) == 0){
            value.push_back(val);
            index[val] = value.size() - 1;
            return true;
        }
        else{
            return false;
        }
    }
    
    //我们不能使用erase来删除vector中的元素，因为这样会导致后面的元素左移，时间复杂度很高
    //而且元素左移会导致map中的所有索引都要改，非常麻烦
    //我们将需要删除的元素换到队尾，用pop_back来删除他，不要忘了把被换走的队尾元素的索引进行更新
    bool remove(int val) {
        if(index.count(val) == 0){
            return false;
        }
        else{
            int delIndex = index[val];
            value[delIndex] = value.back();
            index[value.back()] = delIndex;
            value.pop_back();
            index.erase(val);
            return true;
        }
    }
    
    int getRandom() {
        //vector可以通过这种方式实现随机访问
        int randomIndex = rand()%value.size();
        return value[randomIndex];
    }

private:
    //为了使随机访问的时间复杂度达到O(1)，需要用到vector；vector用于存储数据
    vector<int> value;

    //为了使插入删除的时间复杂度达到O(1)，需要用到map；map用于存储值和vector下标的映射
    unordered_map<int, int> index;
};
```
