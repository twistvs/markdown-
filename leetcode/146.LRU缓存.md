<u>相当复杂的一个新建数据结构问题</u>

------

​	**如何记录关键字的常用顺序？**

容易想到list，最新访问的关键字放到队尾，如果capacity已满，就把访问的最少的队头弹出

​	**如果不用list，该怎么实现？**

list本质上就是个双向链表，所以我们要自己创建一个双向链表

------

1. 创建节点数据结构Listnode，成员应该包括key,val,prev,next
2. 创建unordered_map<key, Listnode*>  实现key值到实际节点的映射

```c++
struct Listnode{
    int key;
    int val;
    Listnode* prev;
    Listnode* next;

    Listnode(int key, int val){
        this->key = key;
        this->val = val;
        prev = NULL;
        next = NULL;
    }
};

class LRUCache {
public:
    //容量，key与节点的映射，双向链表的头节点和尾节点
    int capacity;
    unordered_map<int, Listnode*> mp;
    Listnode* head = new Listnode(0, 0);
    Listnode* tail = new Listnode(0, 0);

    //初始化，获得capacity的值，并且将head和tail连接起来
    LRUCache(int capacity) {
        this->capacity = capacity;
        head->next = tail;
        tail->prev = head;
    }
    
    int get(int key) {
        //如果没找到，返回-1
        if(mp.count(key) == 0){
            return -1;
        }
        //如果找到了，返回节点的值，并且将节点放到队尾
        else{
            moveToTail(mp[key]);
            return mp[key]->val;
        }
    }
    
    void put(int key, int value) {
        //如果map里面已经有这个key了，更新val值并且把key放到队尾
        if(mp.count(key)){
            mp[key]->val = value;
            moveToTail(mp[key]);
        }
        //如果map里面没有这个key，把key插入队尾
        else{
            Listnode* node = new Listnode(key, value);
            mp[key] = node;
            addToTail(node);
            
            //如果插入进来之后容量不够用了，要弹出队头
            if(mp.size() > capacity){
                Listnode* erased = popHead();
                mp.erase(erased->key);
                delete erased;
            }
        }
    }
	
    //挪到队尾的操作可以分解成删除后再添加到队尾
    void moveToTail(Listnode* node){
        deleteNode(node);
        addToTail(node);
    }

    void deleteNode(Listnode* node){
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }

    void addToTail(Listnode* node){
        tail->prev->next = node;
        node->prev = tail->prev;
        node->next = tail;
        tail->prev = node;
    }

    //由于我们要用key来访问map以删除map的对应项，所以需要把弹出的队头返回去
    Listnode* popHead(){
        Listnode* temp = head->next;
        deleteNode(temp);
        return temp;
    }
};
```
