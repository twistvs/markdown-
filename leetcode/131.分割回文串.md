#### 动态规划+回溯

- 首先看到需要输出所有方案，必须要用回溯
- 在回溯的过程中，我们需要判断子串是否为回文串来判断路径是否有效，所以需要用dp把这个信息提前存起来

```c++
class Solution {
public:
    vector<vector<string>> ans;
    vector<string> temp;
    int n;

    void backtrack(string& s, vector<vector<bool>> dp, int left){
        if(left == n){
            ans.push_back(temp);
            return;
        }

        //第一选为right的位置，第二选的left为right+1
        for(int right = left; right < n; right++){
            if(dp[left][right]){
                temp.push_back(s.substr(left, right - left + 1));
                backtrack(s, dp, right + 1);
                temp.pop_back();
            }
        }
    }
    vector<vector<string>> partition(string s) {
        n = s.size();

        vector<vector<bool>> dp(n, vector<bool>(n, false));

        //回溯的过程中需要判断子串是否是回文子串来确定路径是否合法，可以提前用dp把子串是否为回文串给存下来
        for(int i = n - 1; i >= 0; i--){
            for(int j = i; j < n; j++){
                if(s[i] == s[j] && (j - i <= 1 || dp[i + 1][j - 1] == 1)){
                    dp[i][j] = true;
                }
            }
        }

        backtrack(s, dp, 0);

        return ans;
    }
};
```
