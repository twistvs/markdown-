#### 堆排序

1. 建立大顶堆，从索引为n/2-1的元素开始，把前面的所有元素都进行一次siftDown
2. 堆排序，堆顶的元素是最大元素，将堆顶元素与nums最后一个元素互换，将大顶堆siftDown，最后的已经排好序的元素就不算在内
3. 对于这道题来说，第k个堆顶就是第k个最大的元素

```c++
class Solution {
public:
    //start是siftDown的起始节点（由数组nums表示），n是nums.size()，即树的结点数
    void siftDown(vector<int>& nums, int start, int n){
        int root = start;
        int left = root * 2 + 1;
        int right = left + 1;
        int largest = root;
        
        while(true){
            //left和right只有在小于n的时候，节点才存在，才进行比较
            if(left < n && nums[left] > nums[largest]){
                largest = left;
            }
            if(right < n && nums[right] > nums[largest]){
                largest = right;
            }
            //如果比较完之后还是root最大，那就不用交换了，这次siftDown结束
            if(largest == root){
                break;
            }
            swap(nums[largest], nums[root]);
            root = largest;
            left = root * 2 + 1;
            right = left + 1;
        }
    }

    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();
        //建立大顶堆，n / 2 - 1是第一个有孩子的节点
        for(int i = n / 2 - 1; i >= 0; i--){
            siftDown(nums, i, n);
        }

        int count = 1;
        //进行堆排序，每次把堆顶的元素与最后一个未排序的元素交换，并重新对堆进行siftDown
        for(int i = n - 1; i >= 0; i--){
            if(count == k){
                return nums[0];
            }
            swap(nums[0], nums[i]);
            siftDown(nums, 0, i);
            count++;
        }
        return 0;
    }
};
```
