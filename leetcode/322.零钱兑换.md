#### 动态规划

- dp[i]表示凑齐i块钱最少需要多少个硬币，他有很多种情况，硬币有多少种面值，他就有多少种情况；因为最后一个硬币的面值可以为所有面值中的任意一个

- 我们在创建dp的时候，可以赋一个极大值，如果在遍历结束后，dp[amount]还是这个极大值，说明amount无法被凑出来，返回-1

```c++
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        //给dp赋极大值
        vector<int> dp(amount + 1, amount + 1);
        if(amount == 0) return 0;
        dp[0] = 0;

        for(int i = 1; i <=amount; i++){
            for(int coin : coins){
                if(i >= coin){
                    //状态转移
                    dp[i] = min(dp[i], dp[i - coin] + 1);
                }
            }
        }
        //如果dp[amount]还是原来的极大值，说明amount无法被凑出来
        if(dp[amount] == amount + 1) return -1;
        return dp[amount];
    }
};
```
