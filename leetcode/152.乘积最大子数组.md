#### 动态规划

这个题并不能直接用`dp[i] = max(dp[i - 1] * nums[i], nums[i])`

- 对于i号元素来讲，如果他是个负数，那么他的最大乘积并不是i-1的最大乘积乘自己，而是i-1的最小乘积乘自己，所以只记录最大乘积是不行的

------

1. 记录最小乘积minValue
2. 记录最大乘积maxValue
3. 同时更新minValue和maxValue

```c++
class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n = nums.size();
        if(n == 0) return 0;
        if(n == 1) return nums[0];
        vector<int> minValue(n);
        vector<int> maxValue(n);
        minValue[0] = nums[0];
        maxValue[0] = nums[0];
        int ans = nums[0];

        for(int i = 1; i < n; i++){
            minValue[i] = min({nums[i], minValue[i - 1]*nums[i], maxValue[i - 1]*nums[i]});
            maxValue[i] = max({nums[i], minValue[i - 1]*nums[i], maxValue[i - 1]*nums[i]});

            ans = max(ans, maxValue[i]);
        }
        return ans;
    }
};
```
