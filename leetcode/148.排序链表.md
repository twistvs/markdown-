#### 链表的归并排序

------

**先讨论一下常规的归并排序**

```c++
vector mergesort(数组){
   //寻找中点，并且用中点把数组分为左右两边
        
   left = mergesort(数组的左半边)；
   right = mergesort(数组的右半边)；
        
   return merge(left, right);
}

vector merge(left, right){
    //创建一个temp用于存放合并后的数组
    vector temp;
    
    while(两个数组都不为空){
        if(left小) 插left；
        if(right小) 插right;
	}
    if(left还有剩) 把剩下的插到后面
    if(right还有剩)  把剩下的插到后面
        
    return temp;
}
```

对于常规的归并排序，我们可以通过size()/2的方式来寻找mid，并且通过l,mid,r三个参数来动态记录数组的左右边界

------

**链表的归并排序**

- 链表没有右边界
- 链表无法通过size()/2来计算mid，只能通过快慢指针来找到mid

```c++
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if(head == NULL || head->next == NULL){
            return head;
        }

        //快慢指针找mid，fast要初始化为head->next
        ListNode* slow = head;
        ListNode* fast = head->next;

        while(fast && fast->next){
            slow = slow->next;
            fast = fast->next->next;
        }

        //将右半部分的头作为mid，然后把左半部分的尾slow断开，这样就分成两部分了
        ListNode* mid = slow->next;
        slow->next = NULL;

        //分别给左右两部分做归并排序
        ListNode* left = sortList(head);
        ListNode* right = sortList(mid);

        //将左右两部分合并然后返回
        return merge(left, right);
    }

    ListNode* merge(ListNode* left, ListNode* right){
        //新建一个链表的标准操作就是先搞一个dummy
        ListNode* dummy = new ListNode(0);
        ListNode* prev = dummy;

        while(left && right){
            //如果left的值更小，那就先插left的
            if(left->val <= right->val){
                prev->next = left;
                prev = prev->next;
                left = left->next;
            }
            else{
                prev->next = right;
                prev = prev->next;
                right = right->next;
            }
        }
		
        //如果其中有一个已经插完了，那就把另一个剩下的部分插到后面
        if(left) prev->next = left;
        if(right) prev->next = right;
        return dummy->next;
    }
};
```

#### merge数组和merge链表的区别

- 数组在其中一个插完之后，另一个剩下的部分还是需要用遍历的方法插入，所以while的条件为p1 < n1 || p2 < n2，在循环内部再额外判断p1和p2是否已经遍历完

```c++
class Solution {
public:
    string mergeAlternately(string word1, string word2) {
        int n1 = word1.size();
        int n2 = word2.size();

        string ans;

        int p1 = 0, p2 = 0;

        while(p1 < n1 || p2 < n2){
            //如果p1已经遍历完了，这个就不会执行，只会继续遍历p2
            if(p1 < n1){
                ans.push_back(word1[p1]);
                p1++;
            }

            if(p2 < n2){
                ans.push_back(word2[p2]);
                p2++;
            }
        }

        return ans;
    }
};
```

- 链表在其中一个遍历完之后，只需要把另一个插到后面即可，并不需要继续遍历，所以while的条件是left && right，只要有一个遍历完了，就在循环外把没有遍历完的那一个插到后面

```c++
ListNode* merge(ListNode* left, ListNode* right) {
      ListNode dummy(0); // 在栈上开辟，比 new 更快且不需要手动 delete
      ListNode* temp = &dummy;
        
      while (left && right) {
          if (left->val < right->val) {
              temp->next = left;
              left = left->next;
          } else {
              temp->next = right;
              right = right->next;
          }
          temp = temp->next;
      }
      // 剩余部分直接连接，不需要 while 循环
      temp->next = left ? left : right;
      return dummy.next;
  }
```
