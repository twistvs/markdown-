#### 链表的归并排序

------

**先讨论一下常规的归并排序**

```c++
vector mergesort(数组){
   //寻找中点，并且用中点把数组分为左右两边
        
   left = mergesort(数组的左半边)；
   right = mergesort(数组的右半边)；
        
   return merge(left, right);
}

vector merge(left, right){
    //创建一个temp用于存放合并后的数组
    vector temp;
    
    while(两个数组都不为空){
        if(left小) 插left；
        if(right小) 插right;
	}
    if(left还有剩) 把剩下的插到后面
    if(right还有剩)  把剩下的插到后面
        
    return temp;
}
```

对于常规的归并排序，我们可以通过size()/2的方式来寻找mid，并且通过l,mid,r三个参数来动态记录数组的左右边界

------

**链表的归并排序**

- 链表没有右边界
- 链表无法通过size()/2来计算mid，只能通过快慢指针来找到mid

```c++
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if(head == NULL || head->next == NULL){
            return head;
        }

        //快慢指针找mid，fast要初始化为head->next
        ListNode* slow = head;
        ListNode* fast = head->next;

        while(fast && fast->next){
            slow = slow->next;
            fast = fast->next->next;
        }

        //将右半部分的头作为mid，然后把左半部分的尾slow断开，这样就分成两部分了
        ListNode* mid = slow->next;
        slow->next = NULL;

        //分别给左右两部分做归并排序
        ListNode* left = sortList(head);
        ListNode* right = sortList(mid);

        //将左右两部分合并然后返回
        return merge(left, right);
    }

    ListNode* merge(ListNode* left, ListNode* right){
        //新建一个链表的标准操作就是先搞一个dummy
        ListNode* dummy = new ListNode(0);
        ListNode* prev = dummy;

        while(left && right){
            //如果left的值更小，那就先插left的
            if(left->val <= right->val){
                prev->next = left;
                prev = prev->next;
                left = left->next;
            }
            else{
                prev->next = right;
                prev = prev->next;
                right = right->next;
            }
        }
		
        //如果其中有一个已经插完了，那就把另一个剩下的部分插到后面
        if(left) prev->next = left;
        if(right) prev->next = right;
        return dummy->next;
    }
};
```
