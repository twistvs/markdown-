#### 利用数组元素记录状态，以达到空间复杂度O(1)

- 我们在遍历的过程中，不能破坏元素的状态，只能在遍历完之后，一次性替换所有元素的状态

- 我们可以加入2和3，2代表死变活，3代表活变死，这两种新的状态并不好破坏元素的原本状态

```c++
class Solution {
public:
    //计算元素周围八个元素有几个是活的
    int liveNum(vector<vector<int>>& board, int i, int j) {
        int m = board.size();
        int n = board[0].size();

        int count = 0;
        if (j > 0) {
            //1和3都代表细胞曾经是活的
            if (board[i][j - 1] == 1 || board[i][j - 1] == 3) {
                count++;
            }
        }

        if (j < n - 1) {
            if (board[i][j + 1] == 1 || board[i][j + 1] == 3) {
                count++;
            }
        }

        if (i > 0) {
            if (j > 0) {
                if (board[i - 1][j - 1] == 1 || board[i - 1][j - 1] == 3) {
                    count++;
                }
            }
            if (board[i - 1][j] == 1 || board[i - 1][j] == 3) {
                count++;
            }
            if (j < n - 1) {
                if (board[i - 1][j + 1] == 1 || board[i - 1][j + 1] == 3) {
                    count++;
                }
            }
        }

        if (i < m - 1) {
            if (j > 0) {
                if (board[i + 1][j - 1] == 1 || board[i + 1][j - 1] == 3) {
                    count++;
                }
            }
            if (board[i + 1][j] == 1 || board[i + 1][j] == 3) {
                count++;
            }
            if (j < n - 1) {
                if (board[i + 1][j + 1] == 1 || board[i + 1][j + 1] == 3) {
                    count++;
                }
            }
        }
        return count;
    }

    void gameOfLife(vector<vector<int>>& board) {
        int m = board.size();
        int n = board[0].size();

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int live_num = liveNum(board, i, j);
                if (board[i][j] == 1) {
                    if (live_num < 2 || live_num > 3) {
                        board[i][j] = 3;
                    }
                } else {
                    if (live_num == 3) {
                        board[i][j] = 2;
                    }
                }
            }
        }

        //一次性更新所有细胞的状态
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 2) {
                    board[i][j] = 1;
                }
                if (board[i][j] == 3) {
                    board[i][j] = 0;
                }
            }
        }
    }
};
```

**优化：对liveNum函数的优化**

- 可以使用方向数组来便捷地进行周围细胞的遍历

```c++
int liveNum(vector<vector<int>>& board, int i, int j) {
    int m = board.size();
    int n = board[0].size();
    int count = 0;

    // 定义 8 个方向的横纵坐标偏移量
    // 分别对应: 左上, 上, 右上, 左, 右, 左下, 下, 右下
    int dx[8] = {-1, -1, -1,  0, 0,  1, 1, 1};
    int dy[8] = {-1,  0,  1, -1, 1, -1, 0, 1};

    for (int k = 0; k < 8; k++) {
        int new_x = i + dx[k];
        int new_y = j + dy[k];

        // 统一检查边界条件，防止越界
        if (new_x >= 0 && new_x < m && new_y >= 0 && new_y < n) {
            // 状态 1 是原本活的，状态 3 是活变死的（原本也是活的）
            if (board[new_x][new_y] == 1 || board[new_x][new_y] == 3) {
                count++;
            }
        }
    }
    return count;
}
```
