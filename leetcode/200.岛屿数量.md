- 由于结果==并不要求grid矩阵保持原样==，所以完全可以采用破坏性的方式来寻找岛屿

1. 先遍历矩阵，找到字符'1'便对这个元素进行dfs，把和他相连的所有元素变成'0'，这样我们就清除了一个岛屿，ans++
2. 由于我们在dfs的过程中把该岛屿全部变成了'0'，所以在后续遍历的过程中，不需要担心重复访问同一个岛屿的问题

```c++
class Solution {
public:
    

    void dfs(vector<vector<char>>& grid, int starti, int startj, int row, int col){
        if(grid[starti][startj] == '0'){
            return;
        }

        grid[starti][startj] = '0';
        if(starti > 0) dfs(grid, starti - 1, startj, row, col);
        if(starti < row - 1) dfs(grid, starti + 1, startj, row, col);
        if(startj > 0) dfs(grid, starti, startj - 1, row, col);
        if(startj < col - 1) dfs(grid, starti, startj + 1, row, col);
    }


    int numIslands(vector<vector<char>>& grid) {
        int row = grid.size();
        int col = grid[0].size();
        int ans = 0;
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(grid[i][j] == '1'){
                    dfs(grid, i, j, row, col);
                    ans++;
                }
            }
        }
        return ans;
        
    }
};
```
