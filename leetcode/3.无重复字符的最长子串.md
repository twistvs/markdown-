#### 滑动窗口

- 这道题的right是一个一个的后移的，不存在连续后移或跳跃的情况，可以用for来写right的循环

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> visited;
        int n = s.size();

        int left = 0;

        int maxLength = 0;

        for(int right = 0; right < n; right++){
            //如果s[right]已经在窗口里了，就从左开始循环挪窗口，直到挪到s[right]
            while(visited.count(s[right])){
                visited.erase(s[left]);
                left++;
            }

            visited.insert(s[right]);
            maxLength = max(maxLength, right - left + 1);
        }
        return maxLength;
    }
};
```

**效率更高的方法，用vector(128)来存储所有acsii字符**

```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n = s.size();
        if (n == 0) {
            return 0;
        }
        
        // 修正 1：ASCII 字符集包含 128 个字符，直接开辟 128 的空间
        vector<bool> visited(128, false);
        int maxLen = 0; // 建议初始化为 0，虽然前面有 n==0 的特判，但更为严谨

        int left = 0;
        int right = 0;

        for (right = 0; right < n; right++) {
            // 修正 2：直接用字符 s[right] 作为下标，不需要减去 'a'
            if (visited[s[right]]) {
                // 如果发现重复，左指针开始向右“追赶”，并沿途清除访问记录
                while (s[left] != s[right]) {
                    visited[s[left]] = false;
                    left++;
                }
                // 左指针跨过那个导致重复的字符
                left++;
            }

            // 标记当前字符已访问
            visited[s[right]] = true;
            // 更新最大长度
            maxLen = max(maxLen, right - left + 1);
        }

        return maxLen;
    }
};
```
