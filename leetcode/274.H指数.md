排序后，至少有1篇论文被引用了citations[n - 1]次，至少有2篇论文被引用了citations[n - 2]次，这样一来，每个元素的篇数和引用数都有了，就可以一个个作比较了

```c++
class Solution {
public:
    int hIndex(vector<int>& citations) {
        int n = citations.size();
        int h = 0;

        int count = 0;

        //排序后，至少有1篇论文被引用了citations[n - 1]次，至少有2篇论文被引用了citations[n - 2]次
        sort(citations.begin(), citations.end());

        for(int i = n - 1; i >= 0; i--){
            count++;
            int temp = min(count, citations[i]);
            h = max(h, min(n, temp));
        }

        return h;
    }
};
```
