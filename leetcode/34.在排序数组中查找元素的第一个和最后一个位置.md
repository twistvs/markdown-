#### 特殊的二分

一般情况下，二分只要nums[mid] == target就会得到结果，但是这里不一样，如果要找第一个，那么在nums[mid] == target时还要继续查找左半部分，如果要找最后一个，那么在nums[mid] == target时还要继续查找右半部分



- 可以把两种情况写在一起，但是比较复杂；用两次二分分别查找第一个和最后一个比较直观

```c++
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n - 1;
        int first = -1;
        int last = -1;

        //先找左边界
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                first = mid;
                right = mid - 1;
            }
            else if(nums[mid] > target){
                right = mid - 1;
            }
            else{
                left = mid + 1;
            }
        }

        //再找右边界
        left = 0;
        right = n - 1;
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                last = mid;
                left = mid + 1;
            }
            else if(nums[mid] > target){
                right = mid - 1;
            }
            else{
                left = mid + 1;
            }
        }
        vector<int> ans = {first, last};
        return ans;
    }
};
```

- 如果采用和寻找插入位置类似的思路的话，也可以解决这道题

```c++
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int n = nums.size();
        vector<int> ans(2, -1);

        int left = 0;
        int right = n - 1;

        //不设置结束条件而是让循环一直执行到自然退出，那么最终的left或right一定会留在夹住target的位置的的旁边，这是找插入位置那道题的思路
        //在这道题中，如果nums包含target，就一定可以找到，通过target <= nums[mid]的条件，可以让left最终稳定的落在最左边的target上
        //如果不包含target，则left或right会落在夹住target的位置的旁边，这可能会导致left超出右边界或者right超出左边界，这两种情况需要验证
        while (left <= right) {
            int mid = (left + right) / 2;
            if (target <= nums[mid]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }

        //验证left < n，并且检查最终left是否落在target上，如果nums包含target，就一定会落在target上
        if (left < n && nums[left] == target) {
            ans[0] = left;
        }

        left = 0;
        right = n - 1;

        while (left <= right) {
            int mid = (left + right) / 2;
            if (target >= nums[mid]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        if (right >= 0 && nums[right] == target) {
            ans[1] = right;
        }

        return ans;
    }
};
```
