- **二叉搜索树的中序遍历是递增的**
- 中序遍历第k个元素就是第k小的元素

```c++
class Solution {
public:
    int ans = 0;

    void zhongxv(TreeNode* root, int& k){
        if(root == NULL){
            return;
        }

        zhongxv(root->left, k);
        //可以直接用k来记录次数，没有必要再加一个count
        k = k - 1;
        if(k == 0){
            ans = root->val;
        }

        zhongxv(root->right, k);

    }

    int kthSmallest(TreeNode* root, int k) {
        zhongxv(root, k);
        return ans;
    }
};
```
