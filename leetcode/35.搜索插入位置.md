#### 特殊的二分查找

- 这里的二分查找除了找相等的元素外，如果没有相等元素还需要找到插入位置
- 可以在每个mid的左右比较一下是否被夹在中间，这样可以确认位置
- 判断的时候会用到mid + 1和mid - 1，这在左右边界是用不了的

```c++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        //由于nums是有序的，所以target大于最后一个或者小于第一个时，都可以直接确定插入位置
        if(target > nums[n - 1]){
            return n;
        }
        if(target < nums[0]){
            return 0;
        }

        int left = 0;
        int right = n - 1;

        //在普通的二分查找之外加入了对mid + 1和mid - 1的判断，看看是否被夹在中间的位置
        while(right >= left){
            int mid = (right + left)/2;
            if(target == nums[mid]){
                return mid;
            }
            if(target > nums[mid]){
                if(mid < n - 1 && target < nums[mid + 1]){
                    return mid + 1;
                }
                else{
                    left = mid + 1;
                }
            }
            if(target < nums[mid]){
                if(mid > 0 && target > nums[mid - 1]){
                    return mid;
                }
                else{
                    right = mid - 1;
                }
            }
        }
        return 0;
    }
};
```
