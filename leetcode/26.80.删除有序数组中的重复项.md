**纯靠背**

删除有序数组中的重复项，使每个项最多出现n次；n是多少，模板里的2就改成多少

```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n = nums.size();
        if (n <= 2) return n; // 长度小于等于2，直接返回

        int slow = 2; // slow 指向第三个位置，前两个默认保留
        for (int fast = 2; fast < n; fast++) {
            // 如果当前元素不等于倒数第二个保留的元素
            // 说明它不是第三个重复项
            if (nums[fast] != nums[slow - 2]) {
                nums[slow] = nums[fast];
                slow++;
            }
        }
        return slow;
    }
};
```
