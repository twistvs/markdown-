#### 贪心

- 和45.跳跃游戏Ⅱ有很大的相似处

- 当一个子串中包含a时，那么他的右边界至少为最后一个a的位置，在遍历的过程中动态更新end，只有遍历追上end时，才进行一次切割

```c++
class Solution {
public:
    vector<int> partitionLabels(string s) {
        //统计每个字符最后出现的位置
        vector<int> pos(26, 0);
        int n = s.size();
        
        vector<int> ans;

        for(int i = 0; i < n; i++){
            pos[s[i] - 'a'] = i;
        }

        int start = 0;//当前区间的起始下标
        int end = 0;//当前区间的最远边界
        for(int i = 0; i < n; i++){
            //动态更新最远边界
            end = max(end, pos[s[i] - 'a']);

			//当遍历追上end时，进行一次切割
            if(i == end){
                ans.push_back(end - start + 1);
                start = end + 1;
            }
            
        }

        return ans;
    }
};
```
