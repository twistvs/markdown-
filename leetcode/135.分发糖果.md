#### 两次遍历

```c++
class Solution {
public:
    int candy(vector<int>& ratings) {
        int n = ratings.size();

        //每个人最少一个糖果
        vector<int> candyNum(n, 1);

        //从左往右遍历
        for(int i = 1; i < n; i++){
            if(ratings[i] > ratings[i - 1]){
                candyNum[i] = max(candyNum[i], candyNum[i - 1] + 1);
            }
        }

        //从右往左遍历
        for(int i = n - 2; i >= 0; i--){
            if(ratings[i] > ratings[i + 1]){
                //注意取最大值
                candyNum[i] = max(candyNum[i], candyNum[i + 1] + 1);
            }
        }

        int sum = 0;

        for(int i : candyNum){
            sum += i;
        }
        return sum;
    }
};
```
