- abcde和abc的最长前缀和最长只可能为3，因为abc都没有4和5，所以最长只需要比较最短元素长度次数就可以
- 从第一个元素开始比，如果全部相等就加进ans，如果有不相等的立刻结束，返回ans

```c++
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        string ans;

        int n = strs.size();

        int minSize = strs[0].size();

        for(string s : strs){
            minSize = min(minSize, int(s.size()));
        }

        for(int i = 0; i < minSize; i++){
            char c = strs[0][i];
            bool flag = true;
            for(int j = 0; j < n; j++){
                if(strs[j][i] != c){
                    flag = false;
                }
            }
            if(flag){
                ans.push_back(c);
            }
            else{
                break;
            }
        }

        return ans;
    }
};
```
